package com.aaa.test;

import java.util.List;
import java.util.Scanner;

import org.apache.ibatis.session.SqlSession;
import com.aaa.dao.DepartmentMapper;
import com.aaa.util.MybatisUtil;
import com.aaa.entity.Department;

public class Test {
	static SqlSession openSession=MybatisUtil.getSession(true);
	static DepartmentMapper mapper=openSession.getMapper(DepartmentMapper.class);
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int a=0;
		boolean b=true;
		do{
			System.out.println("请输入要执行的操作：");	
			System.out.println("1.查所有   2.按id查询 3.查看总条数4.添加 5.修改");
			a=sc.nextInt();
			switch(a){
			case 1:
				selectDepartment();
				break;
			case 2:
				selectDepartmentById();
				break;
			case 3:
				selectCount();
				break;
			case 4:
				insertDepartment();
				break;
			default:
				System.out.println("请输入正确的数字指令");
				break;
				
			}
			System.out.println("是否继续?(1.继续2.退出)");
			int num = sc.nextInt();
			if (num == 1) {
				continue;
			}
			if (num == 2) {
				b=false;
				System.out.println("欢迎下次使用");
			}
			
		}while(a!=0);
		
		
		
		
		
	}


public static void selectDepartment(){
	List<Department> selectDepartment=mapper.selectDepartment();
	for(Department d:selectDepartment){
		System.out.println(d);
	}
}
public static void selectDepartmentById(){
	System.out.println("入要查的id");
	Scanner sc = new Scanner(System.in);
	int id=sc.nextInt();
	List<Department> list = mapper.selectDepartmentById(new Department(id));
	for(Department a:list){
		System.out.println(a);
	}
}
public static int selectCount(){
	int i=mapper.selectCountDepartment();
	System.out.println("总条数："+i);
	return i;
	
}
public static void insertDepartment(){
	System.out.println("请输入要插入的名字：");
	Scanner sc = new Scanner(System.in);
	String name=sc.next();
	String descp=sc.next();
	int i = mapper.insertDepartment(new Department(name,descp));
	System.out.println(i);
}
	}
